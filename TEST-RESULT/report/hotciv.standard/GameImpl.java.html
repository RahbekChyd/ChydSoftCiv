public class GameImpl implements Game {
	public Player bluePlayer = Player.BLUE;
	public Player redPlayer = Player.RED;
	public Player playerInTurn = Player.RED;
	public City redCity = new CityImpl(Player.RED);
	public City blueCity = new CityImpl(Player.BLUE);
	public Tile ocean = new TileImpl("Ocean");
	public Tile hill = new TileImpl("Hill");
	public Tile mountain = new TileImpl("Mountain");
	public Tile plain = new TileImpl("Plain");
	public ArrayList<Unit> units = new ArrayList();
	public Unit redArcher = new UnitImpl("archer", Player.RED, new Position(2, 0));
	public Unit blueLegion = new UnitImpl("legion", Player.BLUE, new Position(3, 2));
	public Unit redSettler = new UnitImpl("settler", Player.RED, new Position(4, 3));
	private int currentAge = -4000;
	private int turnCounter = 0;
	
	public GameImpl() {
		units.add(redArcher);
		units.add(blueLegion);
		units.add(redSettler);
	}

	public Tile getTileAt( Position p ) {
		if (p.getRow() == 1 && p.getColumn() == 0)
			return ocean;
		if (p.getRow() == 0 && p.getColumn() == 1)
			return hill;
		if (p.getRow() == 2 && p.getColumn() == 2)
			return mountain;
		return plain; 
	}
	public Unit getUnitAt( Position p ) {
		for (Unit unit : units) {
			if (unit.getPosition().equals(p))
				return unit;
		}
		return null; 
	}
	
	public City getCityAt( Position p ) { 
		if (p.getRow() == 1 && p.getColumn() == 1)
			return redCity;
		if (p.getRow() == 4 && p.getColumn() == 1)
			return blueCity;
		return null; 
	}
	public Player getPlayerInTurn() { 
		return playerInTurn;
		}
	
	public Player getWinner() {
		if (getAge() == -3000)
			return redPlayer;
		return null; 
		}
	
	public int getAge() { 
		return currentAge;
		}
	
	public boolean moveUnit( Position from, Position to ) {
		return false;
	}
	
	public void endOfTurn() {
		if (getPlayerInTurn() == Player.RED)
			playerInTurn = Player.BLUE;
		else {
			playerInTurn = Player.RED;
		}
		turnCounter += 1;
		
		if (turnCounter == 2) {
			redCity.updateProduction();
			blueCity.updateProduction();
			currentAge += 100;
			turnCounter = 0;
		}
	}
	
	public void changeWorkForceFocusInCityAt( Position p, String balance ) {}
	public void changeProductionInCityAt( Position p, String unitType ) {}
	public void performUnitActionAt( Position p ) {}
	
	public boolean isTileAlreadyTaken( Position p ) {
		if (getUnitAt(p) != null)
			return true;
		return false;
	}
}
